
CREATE TABLE UNICREDIT_TRANS_TAB
(
TID_NO NUMBER(10),
ACC_NO NUMBER(20)
CONSTRAINT ACC_NO_FK
REFERENCES UNICREDIT_ACC_TAB(ACC_NO),
TRANS_TYPE CHAR(1)
CONSTRAINT TRANS_TYPE_CHK01
CHECK(TRANS_TYP = TRANS_TYPE IN('D', 'W')),
DATE_OF_TRANS DATE
DEFAULT SYSDATE,
AMOUNT NUMBER(7, 2)
CONSTRAINT AMOUNT_CHK
CHECK(AMOUNT = AMOUNT >= 100)
)
