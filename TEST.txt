 DECLARE
 CURSOR EMPCURSOR
 IS
 SELECT ENAME FROM EMP;
 TYPE EMP_TYPE
 IS TABLE OF EMPCURSOR%ROWTYPE;
 V_EMP_TYPE EMP_TYPE;
 BEGIN
 OPEN EMPCURSOR;
 LOOP
 FETCH EMPCURSOR BULK COLLECT INTO V_EMP_TYPE LIMIT 8;
 FOR INDX IN 1..V_EMP_TYPE.COUNT
 LOOP
 DBMS_OUTPUT.PUT_LINE('EMPLOYEE NAME IS :'||V_EMP_TYPE(INDX).ENAME);
 END LOOP;
 EXIT WHEN EMPCURSOR%NOTFOUND;
 END LOOP;
 END;
/
